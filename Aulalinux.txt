todos os comandos no linux é minusculo



sudo apt update           #atualizar o sources.list

sudo apt upgrade          #atualizar o sistema

sudo apt full-upgrade     #atualização completa do sistema

sudo apt dist-upgrade     #atualização da distribuição

sudo apt autoremove       #auto remoção de softwares

sudo apt autoclean        #auto limpeza do sistema operacional

sudo reboot               #reiniciar o sistema 

history                   #histórico de comandos

sudo apt install vim      #instalar o software
